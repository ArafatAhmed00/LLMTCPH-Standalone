	R_NAME TEXT NOT NULL,  -- size 25

)

Table: NATION
CREATE STATEMENT: CREATE TABLE NATION (
	N_NATIONKEY INTEGER PRIMARY KEY NOT NULL,  -- 25 nations are populated
	N_NAME TEXT NOT NULL,  -- size 25
	N_REGIONKEY INTEGER NOT NULL,  -- Foreign Key to R_REGIONKEY
	N_COMMENT TEXT  -- size 152
	FOREIGN KEY (N_REGIONKEY) REFERENCES REGION(R_REGIONKEY)
)

Table: PART
CREATE STATEMENT: CREATE TABLE PART (
	P_PARTKEY INTEGER PRIMARY KEY NOT NULL,  -- SF*200,000 are populated
	P_NAME TEXT NOT NULL,  -- size 55
	P_MFGR TEXT NOT NULL,  -- size 25
	P_BRAND TEXT NOT NULL,  -- size 10
	P_TYPE TEXT NOT NULL,  -- size 25
	P_SIZE INTEGER NOT NULL,
	P_CONTAINER TEXT NOT NULL,  -- size 10
	P_RETAILPRICE DECIMAL NOT NULL,
	P_COMMENT TEXT NOT NULL  -- size 23
)

Table: SUPPLIER
CREATE STATEMENT: CREATE TABLE SUPPLIER (
	S_SUPPKEY INTEGER PRIMARY KEY NOT NULL,  -- SF*10,000 are populated
	S_NAME TEXT NOT NULL,  -- size 25
	S_ADDRESS TEXT NOT NULL,  -- size 40
	S_NATIONKEY INTEGER NOT NULL,  -- Foreign Key to N_NATIONKEY
	S_PHONE TEXT NOT NULL,  -- size 15
	S_ACCTBAL DECIMAL NOT NULL,
	S_COMMENT TEXT NOT NULL  -- size 101
	FOREIGN KEY (S_NATIONKEY) REFERENCES NATION(N_NATIONKEY)
)

Table: PARTSUPP
CREATE STATEMENT: CREATE TABLE PARTSUPP (
	PS_PARTKEY INTEGER NOT NULL,  -- Foreign Key to P_PARTKEY
	PS_SUPPKEY INTEGER NOT NULL,  -- Foreign Key to S_SUPPKEY
	PS_AVAILQTY INTEGER NOT NULL,
	PS_SUPPLYCOST DECIMAL NOT NULL,
	PS_COMMENT TEXT NOT NULL,  -- size 199
	PRIMARY KEY (PS_PARTKEY, PS_SUPPKEY),
	FOREIGN KEY (PS_SUPPKEY) REFERENCES SUPPLIER(S_SUPPKEY),
	FOREIGN KEY (PS_PARTKEY) REFERENCES PART(P_PARTKEY)
)

Table: CUSTOMER
CREATE STATEMENT: CREATE TABLE CUSTOMER (
	C_CUSTKEY INTEGER PRIMARY KEY NOT NULL,  -- SF*150,000 are populated
	C_NAME TEXT NOT NULL,  -- size 25
	C_ADDRESS TEXT NOT NULL,  -- size 40
	C_NATIONKEY INTEGER NOT NULL,  -- Foreign Key to N_NATIONKEY
	C_PHONE TEXT NOT NULL,  -- size 15
	C_ACCTBAL DECIMAL NOT NULL,
	C_MKTSEGMENT TEXT NOT NULL,  -- size 10
	C_COMMENT TEXT NOT NULL  -- size 117
	FOREIGN KEY (C_NATIONKEY) REFERENCES NATION(N_NATIONKEY)
)

Table: ORDERS
CREATE STATEMENT: CREATE TABLE ORDERS (
	O_ORDERKEY INTEGER PRIMARY KEY NOT NULL,  -- SF*1,500,000 are sparsely populated
	O_CUSTKEY INTEGER NOT NULL,  -- Foreign Key to C_CUSTKEY
	O_ORDERSTATUS TEXT NOT NULL,  -- size 1
	O_TOTALPRICE DECIMAL NOT NULL,
	O_ORDERDATE DATE NOT NULL,
	O_ORDERPRIORITY TEXT NOT NULL,  -- size 15
	O_CLERK TEXT NOT NULL,  -- size 15
	O_SHIPPRIORITY INTEGER NOT NULL,
	O_COMMENT TEXT NOT NULL  -- size 79
	FOREIGN KEY (O_CUSTKEY) REFERENCES CUSTOMER(C_CUSTKEY)
)

Table: LINEITEM
CREATE STATEMENT: CREATE TABLE LINEITEM (
	L_ORDERKEY INTEGER NOT NULL,  -- Foreign Key to O_ORDERKEY
	L_PARTKEY INTEGER NOT NULL,  -- Foreign key to P_PARTKEY, first part of the compound Foreign Key to (PS_PARTKEY, PS_SUPPKEY) with L_SUPPKEY
	L_SUPPKEY INTEGER NOT NULL,  -- Foreign key to S_SUPPKEY, second part of the compound Foreign Key to (PS_PARTKEY, PS_SUPPKEY) with L_PARTKEY
	L_LINENUMBER INTEGER NOT NULL,
	L_QUANTITY DECIMAL NOT NULL,
	L_EXTENDEDPRICE DECIMAL NOT NULL,
	L_DISCOUNT DECIMAL NOT NULL,
	L_TAX DECIMAL NOT NULL,
	L_RETURNFLAG TEXT NOT NULL,  -- size 1
	L_LINESTATUS TEXT NOT NULL,  -- size 1
	L_SHIPDATE DATE NOT NULL,
	L_COMMITDATE DATE NOT NULL,
	L_RECEIPTDATE DATE NOT NULL,
	L_SHIPINSTRUCT TEXT NOT NULL,  -- size 25
	L_SHIPMODE TEXT NOT NULL,  -- size 10
	L_COMMENT TEXT NOT NULL,  -- size 44
	PRIMARY KEY (L_ORDERKEY, L_LINENUMBER),
	FOREIGN KEY (L_ORDERKEY) REFERENCES ORDERS(O_ORDERKEY),
	FOREIGN KEY (L_PARTKEY, L_SUPPKEY) REFERENCES PARTSUPP(PS_PARTKEY, PS_SUPPKEY)
)
